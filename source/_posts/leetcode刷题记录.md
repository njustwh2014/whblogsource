public List<String> removeInvalidParentheses(String s) {
    List<String> ret=new ArrayList<>();
    if(s==null||s.length()==0){
        ret.add("");
        return ret;
    }
    removeInvalidParenthesesHelper(s,ret,0,0,new char[]{'(',')'});
    return ret;
}
